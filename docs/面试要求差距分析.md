#### 3. 自动化测试脚本
```python
# test/test_rag.py
"""自动化测试脚本"""
import sys
import os

sys.path.insert(0, os.path.dirname(os.path.dirname(__file__)))

from main import RAGRobot

def test_rag_pipeline():
    """测试完整RAG流程"""
    # 测试用例
    test_cases = [
        {
            "question": "本标准规定了除什么以外的各种键的技术条件？",
            "type": "正文问答",
            "expected": "应回答'除矩形花键以外'",
        },
        {
            "question": "键的抗拉强度要求是多少？",
            "type": "表格问答",
            "expected": "应回答具体数值要求",
        },
        {
            "question": "键的表面粗糙度要求是多少？",
            "type": "无答案问题",
            "expected": "应礼貌拒答",
        },
        {
            "question": "这个标准怎么样？",
            "type": "模糊问题",
            "expected": "应引导具体提问",
        },
        {
            "question": "键宽10mm的平键，抗拉强度要求是多少？",
            "type": "数值比较",
            "expected": "应基于文档数值判断",
        },
    ]
    
    # 执行测试
    robot = RAGRobot(collection_name="rag_robot_main")
    if not robot.load():
        print("索引不存在，开始构建...")
        robot.build()
    
    results = []
    for i, tc in enumerate(test_cases, 1):
        print(f"\n{'='*60}")
        print(f"测试 {i}/{len(test_cases)}: [{tc['type']}]")
        print(f"问题: {tc['question']}")
        print(f"预期: {tc['expected']}")
        print("-" * 60)
        
        result = robot.ask(tc['question'])
        print(result)
        
        results.append({
            "question": tc['question'],
            "type": tc['type'],
            "expected": tc['expected'],
            "result": result,
        })
    
    # 输出测试报告
    print("\n" + "=" * 60)
    print("测试报告")
    print("=" * 60)
    for i, r in enumerate(results, 1):
        print(f"{i}. [{r['type']}] {r['question'][:30]}...")
        print(f"   预期: {r['expected']}")
        print()

if __name__ == "__main__":
    test_rag_pipeline()
```
